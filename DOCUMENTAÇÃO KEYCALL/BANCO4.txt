CREATE TABLE USUARIO(
 IDUSO SERIAL PRIMARY KEY,
 USUTIPO INTEGER NOT NULL,
 USOEMAIL VARCHAR(100) NOT NULL,
 USOCPF VARCHAR(50) NOT NULL
 );

CREATE TABLE FUNCIONARIO(
 IDFUNC SERIAL PRIMARY KEY,
 FUNCNOME VARCHAR(100) NOT NULL,
 FUNCCARGO VARCHAR(100) NOT NULL
 );


 CREATE TABLE CTELEFONICA(
  IDTEL SERIAL PRIMARY KEY,
  CTSOLITANTE VARCHAR(100) NOT NULL, 
  CTSETSOLICITANTE VARCHAR(100) NOT NULL,
  CTRAMAL CHAR(3) NOT NULL,
  CTNUMERO CHAR(20) NOT NULL, 
  CTHORARIO DATE NOT NULL);

  CREATE TABLE CHAVE(
   IDCHAVE SERIAL PRIMARY KEY,
   CHDATARETIRADA DATE NOT NULL,
   CHSALA VARCHAR(100) NOT NULL,
   CHHORARIORE VARCHAR(100) NOT NULL,
   CHDATAENTRADA DATE NOT NULL,
   CHHORARIOEN VARCHAR(100) NOT NULL
   CHRESPONSAVELRE VARCHAR(100) NOT NULL

  );

  CREATE TABLE MINI(
   IDMINI SERIAL PRIMARY KEY,
   MINDATA DATE NOT NULL, 
   MINHORARIO VARCHAR(15) NOT NULL,
   MINRESPONSAVEL VARCHAR(100) NOT NULL

  );

  CREATE TABLE VEICULOS(
  IDVEI SERIAL PRIMARY KEY,
  TIPOVEI INTEGER NOT NULL,
  DATASAI DATE NOT NULL, 
  HORARIOSAI VARCHAR(100) NOT NULL,
  VEICULO VARCHAR(50) NOT NULL,
  PLACA VARCHAR(50) NOT NULL,
  DESTINO VARCHAR(100) NOT NULL,
  HODOMETROSAI VARCHAR(100) NOT NULL,
  MOTORISTA VARCHAR(100) NOT NULL,
  VIGILANTE VARCHAR(100) NOT NULL
);
